# TransactionHistory (`cui-transaction-history`)

## Example

### JavaScript

```html
<div id="transaction-history-container"></div>
<script>
  const TransactionHistoryEl = document.createElement("cui-transaction-history");
  const historyItems = [
    {
      created_at: "2018-07-20T09:10:14+00:00",
      description: "Target #9283 Atlanta",
      line_item_status: "VALID",
      original_amount_cents: 84000,
    },
    {
      created_at: "2018-08-20T09:10:14+00:00",
      description: "Target #9283 Atlanta",
      line_item_status: "VALID",
      original_amount_cents: 18325,
    },
    {
      created_at: "2018-09-20T09:10:14+00:00",
      description: "Target #9283 Atlanta",
      line_item_status: "VALID",
      original_amount_cents: 18325,
    },
  ];
  TransactionHistoryEl.setAttribute("items", JSON.stringify(historyItems));
  document.getElementById("transaction-history-container").appendChild(TransactionHistoryEl);
</script>
```
