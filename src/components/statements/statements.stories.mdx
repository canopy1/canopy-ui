# Statements (`cui-statements`)

List of Canopy statements. Clicking one will open an embedded PDF preview for it.

_NOTE: This is an experimental component and may be subject to change._

## Example

### JavaScript

```html
<div id="statements-container"></div>

<script>
  const statementsEl = document.createElement("cui-statements");
  const statements = [
    {
      statement_id: 1,
      cycle_inclusive_start: "2021-05-01T09:42:00+00:00",
      cycle_exclusive_end: "2021-05-08T09:42:00+00:00",
      statement_pdf_url: "https://www.w3.org/WAI/ER/tests/xhtml/testfiles/resources/pdf/dummy.pdf",
    },
    {
      statement_id: 2,
      cycle_inclusive_start: "2021-05-08T09:42:00+00:00",
      cycle_exclusive_end: "2021-05-15T09:42:00+00:00",
      statement_pdf_url: "https://www.w3.org/WAI/ER/tests/xhtml/testfiles/resources/pdf/dummy.pdf",
    },
    // ...
  ];
  statementsEl.setAttribute("statements", JSON.stringify(statements));
  document.getElementById("statements-container").appendChild(statementsEl);
</script>
```
