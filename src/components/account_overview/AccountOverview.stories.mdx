import { html } from "lit";
import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import { AccountOverview } from "./account_overview";

<Meta title="Components/AccountOverview" component="cui-account-overview" />

export const Default = ({ total_balance_cents, credit_limit_cents, available_credit_cents, pending_charges }) => {
  const details = {
    total_balance_cents,
    credit_limit_cents,
    available_credit_cents,
    pending_charges,
  };
  return html`
    <cui-account-overview details=${JSON.stringify(details)} style="min-width: 900px;"> </cui-account-overview>
  `;
};

# AccountOverview (`cui-account-overview`)

## Example

### HTML

```html
<cui-account-overview
  details='{
    "total_balance_cents": 349392,
    "credit_limit_cents": 800000,
    "available_credit_cents": 450608,
    "pending_charges": 0,
  }'
>
</cui-account-overview>
```

### JavaScript

```html
<div id="account-overview-container"></div>

<script>
  const accountOverviewEl = document.createElement("cui-account-overview");
  const accountDetails = {
    total_balance_cents: 349391,
    credit_limit_cents: 800000,
    available_credit_cents: 450608,
    pending_charges: 0,
  };
  accountOverviewEl.setAttribute("details", JSON.stringify(accountDetails));
  document.getElementById("account-overview-container").appendChild(accountOverviewEl);
</script>
```
